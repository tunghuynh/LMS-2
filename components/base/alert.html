Code Example

<!-- Basic Alert -->
<div class="alert alert--success">
    <div class="alert__icon">
        <i class="fas fa-check-circle"></i>
    </div>
    <div class="alert__content">
        <p>Success! Your changes have been saved.</p>
    </div>
</div>

<!-- Alert with Title -->
<div class="alert alert--warning">
    <div class="alert__icon">
        <i class="fas fa-exclamation-triangle"></i>
    </div>
    <div class="alert__content">
        <h4 class="alert__title">Warning</h4>
        <p>Please review your input.</p>
    </div>
</div>

<!-- Dismissible Alert -->
<div class="alert alert--info alert--dismissible">
    <div class="alert__icon">
        <i class="fas fa-info-circle"></i>
    </div>
    <div class="alert__content">
        <p>This alert can be dismissed.</p>
    </div>
    <button class="alert__close" aria-label="Close">
        <i class="fas fa-times"></i>
    </button>
</div>

<!-- Alert with Actions -->
<div class="alert alert--danger">
    <div class="alert__icon">
        <i class="fas fa-times-circle"></i>
    </div>
    <div class="alert__content">
        <h4 class="alert__title">Confirm Delete</h4>
        <p>This action cannot be undone.</p>
        <div class="alert__actions">
            <button class="btn btn--danger btn--small">Delete</button>
            <button class="btn btn--secondary btn--small">Cancel</button>
        </div>
    </div>
</div>
